grep -r lukasz /etc/* 2> /dev/null
grep -r `whoami` /etc/* 2> /dev/null

cat numbers01 > numbers_all
cat numbers02 >> numbers_all
cat numbers03 >> numbers_all
sort -n -f numbers_all > numbers
